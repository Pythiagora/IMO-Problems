\textbf{Problem 1.}
Determine all three-digit numbers $N$ having the property that $N$ is divisible by 11, and $N/11$ is equal to the sum of the squares of the digits of $N$.

\bigskip

\textbf{Problem 2.}
For what values of the variable $x$ does the following inequality hold:
\[
\frac{4x^2}{\left(1 - \sqrt{1+2x}\right)^2} < 2x + 9\,?
\]

\bigskip

\textbf{Problem 3.}
In a given right triangle $ABC$, the hypotenuse $BC$, of length $a$, is divided into $n$ equal parts ($n$ an odd integer). Let $\alpha$ be the acute angle subtending, from $A$, that segment which contains the midpoint of the hypotenuse. Let $h$ be the length of the altitude to the hypotenuse of the triangle. Prove:
\[
\tan\alpha = \frac{4nh}{(n^2-1)a}.
\]

\bigskip

\textbf{Problem 4.}
Construct triangle $ABC$, given $h_a$, $h_b$ (the altitudes from $A$ and $B$) and $m_a$, the median from vertex $A$.

\bigskip

\textbf{Problem 5.}
Consider the cube $ABCDA'B'C'D'$ (with face $ABCD$ directly above face $A'B'C'D'$).
\begin{enumerate}
\item[(a)] Find the locus of the midpoints of segments $XY$, where $X$ is any point of $AC$ and $Y$ is any point of $B'D'$.
\item[(b)] Find the locus of points $Z$ which lie on the segments $XY$ of part (a) with $ZY = 2XZ$.
\end{enumerate}

\bigskip

\textbf{Problem 6.}
Consider a cone of revolution with an inscribed sphere tangent to the base of the cone. A cylinder is circumscribed about this sphere so that one of its bases lies in the base of the cone. Let $V_1$ be the volume of the cone and $V_2$ the volume of the cylinder.
\begin{enumerate}
\item[(a)] Prove that $V_1 \neq V_2$.
\item[(b)] Find the smallest number $k$ for which $V_1 = kV_2$; for this case, construct the angle subtended by a diameter of the base of the cone at the vertex of the cone.
\end{enumerate}

\bigskip

\textbf{Problem 7.}
An isosceles trapezoid with bases $a$ and $c$ and altitude $h$ is given.
\begin{enumerate}
\item[(a)] On the axis of symmetry of this trapezoid, find all points $P$ such that both legs of the trapezoid subtend right angles at $P$.
\item[(b)] Calculate the distance of $P$ from either base.
\item[(c)] Determine under what conditions such points $P$ actually exist. (Discuss various cases that might arise.)
\end{enumerate}
