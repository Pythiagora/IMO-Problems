\textbf{Problem 1.}
$P$ is a point inside a given triangle $ABC$. $D$, $E$, $F$ are the feet of the perpendiculars from $P$ to the lines $BC$, $CA$, $AB$ respectively. Find all $P$ for which
\[
\frac{BC}{PD} + \frac{CA}{PE} + \frac{AB}{PF}
\]
is least.

\bigskip

\textbf{Problem 2.}
Let $1 \leq r \leq n$ and consider all subsets of $r$ elements of the set $\{1, 2, \ldots, n\}$. Each of these subsets has a smallest member. Let $F(n, r)$ denote the arithmetic mean of these smallest numbers; prove that
\[
F(n, r) = \frac{n+1}{r+1}.
\]

\bigskip

\textbf{Problem 3.}
Determine the maximum value of $m^3 + n^3$, where $m$ and $n$ are integers satisfying $m, n \in \{1, 2, \ldots, 1981\}$ and $(n^2 - mn - m^2)^2 = 1$.

\bigskip

\textbf{Problem 4.}
\begin{enumerate}
    \item[(a)] For which values of $n > 2$ is there a set of $n$ consecutive positive integers such that the largest number in the set is a divisor of the least common multiple of the remaining $n - 1$ numbers?
    \item[(b)] For which values of $n > 2$ is there exactly one set having the stated property?
\end{enumerate}

\bigskip

\textbf{Problem 5.}
Three congruent circles have a common point $O$ and lie inside a given triangle. Each circle touches a pair of sides of the triangle. Prove that the incenter and the circumcenter of the triangle and the point $O$ are collinear.

\bigskip

\textbf{Problem 6.}
The function $f(x, y)$ satisfies
\begin{enumerate}
    \item[(1)] $f(0, y) = y + 1$,
    \item[(2)] $f(x + 1, 0) = f(x, 1)$,
    \item[(3)] $f(x + 1, y + 1) = f(x, f(x + 1, y))$,
\end{enumerate}
for all non-negative integers $x, y$. Determine $f(4, 1981)$.
