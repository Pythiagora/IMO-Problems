\textbf{Problem 1.}
The function $f(n)$ is defined for all positive integers $n$ and takes on non-negative integer values. Also, for all $m, n$
\[
f(m + n) - f(m) - f(n) = 0 \text{ or } 1,
\]
$f(2) = 0$, $f(3) > 0$, and $f(9999) = 3333$. Determine $f(1982)$.

\bigskip

\textbf{Problem 2.}
A non-isosceles triangle $A_1 A_2 A_3$ is given with sides $a_1, a_2, a_3$ ($a_i$ is the side opposite $A_i$). For all $i = 1, 2, 3$, $M_i$ is the midpoint of side $a_i$, and $T_i$ is the point where the incircle touches side $a_i$. Denote by $S_i$ the reflection of $T_i$ in the interior bisector of angle $A_i$. Prove that the lines $M_1 S_1$, $M_2 S_2$, and $M_3 S_3$ are concurrent.

\bigskip

\textbf{Problem 3.}
Consider the infinite sequences $\{x_n\}$ of positive real numbers with the following properties:
\[
x_0 = 1, \quad \text{and for all } i \geq 0,\ x_{i+1} \leq x_i.
\]
\begin{enumerate}
    \item[(a)] Prove that for every such sequence, there is an $n \geq 1$ such that
    \[
    \frac{x_0^2}{x_1} + \frac{x_1^2}{x_2} + \cdots + \frac{x_{n-1}^2}{x_n} \geq 3.999.
    \]
    \item[(b)] Find such a sequence for which
    \[
    \frac{x_0^2}{x_1} + \frac{x_1^2}{x_2} + \cdots + \frac{x_{n-1}^2}{x_n} < 4.
    \]
\end{enumerate}

\bigskip

\textbf{Problem 4.}
Prove that if $n$ is a positive integer such that the equation
\[
x^3 - 3xy^2 + y^3 = n
\]
has a solution in integers $(x, y)$, then it has at least three such solutions. Show that the equation has no solutions in integers when $n = 2891$.

\bigskip

\textbf{Problem 5.}
The diagonals $AC$ and $CE$ of the regular hexagon $ABCDEF$ are divided by the inner points $M$ and $N$, respectively, so that
\[
\frac{AM}{AC} = \frac{CN}{CE} = r.
\]
Determine $r$ if $B$, $M$, and $N$ are collinear.

\bigskip

\textbf{Problem 6.}
Let $S$ be a square with sides of length $100$, and let $L$ be a path within $S$ which does not meet itself and which is composed of line segments $A_0 A_1, A_1 A_2, \ldots, A_{n-1} A_n$ with $A_0 \neq A_n$. Suppose that for every point $P$ of the boundary of $S$ there is a point of $L$ at a distance from $P$ not greater than $1/2$. Prove that there are two points $X$ and $Y$ in $L$ such that the distance between $X$ and $Y$ is not greater than $1$, and the length of that part of $L$ which lies between $X$ and $Y$ is not smaller than $198$.
